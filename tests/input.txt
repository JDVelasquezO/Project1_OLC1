func Suma(int a, int b){
    var suma = a + b
    return suma + 600
}

func Suma5(int a, int b, int c, int d, int e){
    return a +b+c+d+e
}

func factorial(int num){
    if(num == 0){
        return 1
    }
    else{
        return num * factorial(num-1)
    }
}

func par(int nump) {
    if (nump == 0) {
        return true;
    }
    return impar(nump - 1);
}

func impar(int numi) {
    if (numi == 0) {
        return false;
    }
    return par(numi - 1);
}
